let todosFormularios = [];
let formulariosFiltrados = [];
let charts = {}; // Armazena inst√¢ncias dos gr√°ficos

// 1. INICIALIZA√á√ÉO
document.addEventListener('DOMContentLoaded', async function() {
    await initDB();
    await carregarDados();

// 2. CARREGAR DADOS DO SERVIDOR
async function carregarDados() {
    const response = await fetch(`${window.location.origin}/get-dados-relatorios.php`);
    const data = await response.json();
    
    if (data.success) {
        todosFormularios = data.formularios;
        formulariosFiltrados = [...todosFormularios];
    }
}

// 3. POPULAR FILTROS
function popularFiltros() {
    // Extrair valores √∫nicos
    const municipios = new Set();
    const sistemas = new Set();
    
    todosFormularios.forEach(form => {
        municipios.add(form.respostas.municipio);
        sistemas.add(form.respostas.sistema_produtivo);
    });
    
    // Popular selects
    const selectMunicipio = document.getElementById('filtro-municipio');
    municipios.forEach(m => {
        const option = document.createElement('option');
        option.value = m;
        option.textContent = m;
        selectMunicipio.appendChild(option);
    });
}

// 4. APLICAR FILTROS
function aplicarFiltros() {
    const dataInicio = document.getElementById('filtro-data-inicio').value;
    const municipio = document.getElementById('filtro-municipio').value;
    
    formulariosFiltrados = todosFormularios.filter(form => {
        if (dataInicio && form.timestampFim < new Date(dataInicio).toISOString()) {
            return false;
        }
        if (municipio && form.respostas.municipio !== municipio) {
            return false;
        }
        return true;
    });
    
    calcularEstatisticas();
    atualizarGraficos();
}

// 5. CALCULAR ESTAT√çSTICAS
function calcularEstatisticas() {
    const total = formulariosFiltrados.length;
    const municipios = new Set(formulariosFiltrados.map(f => f.respostas.municipio)).size;
    
    document.getElementById('total-formularios').textContent = total;
    document.getElementById('total-municipios').textContent = municipios;
}

// Exemplo: Gr√°fico de Barras (√Årea Produtiva)
function criarGraficoArea() {
    const faixas = {
        '0-5': 0,
        '5-10': 0,
        '10-20': 0,
        '20-50': 0,
        '50+': 0
    };
    
    formulariosFiltrados.forEach(f => {
        const area = parseFloat(f.respostas.area_produtiva);
        if (area < 5) faixas['0-5']++;
        else if (area < 10) faixas['5-10']++;
        else if (area < 20) faixas['10-20']++;
        else if (area < 50) faixas['20-50']++;
        else faixas['50+']++;
    });
    
    const ctx = document.getElementById('chartArea').getContext('2d');
    charts['chartArea'] = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: Object.keys(faixas),
            datasets: [{
                label: 'N√∫mero de Produtores',
                data: Object.values(faixas),
                backgroundColor: '#2e7d32'
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            scales: {
                y: {
                    beginAtZero: true
                }
            }
        }
    });
}

// Exemplo: Gr√°fico de Radar (Satisfa√ß√£o)
function criarGraficoSatisfacao() {
    const aspectos = [
        { key: 'satisfacao_geral', label: 'Geral' },
        { key: 'satisfacao_atendimento', label: 'Atendimento' },
        { key: 'satisfacao_resultado', label: 'Resultado' }
    ];
    
    const medias = aspectos.map(asp => {
        const valores = formulariosFiltrados
            .map(f => parseInt(f.respostas[asp.key]))
            .filter(v => !isNaN(v));
        return (valores.reduce((a, b) => a + b, 0) / valores.length).toFixed(2);
    });
    
    const ctx = document.getElementById('chartSatisfacao').getContext('2d');
    charts['chartSatisfacao'] = new Chart(ctx, {
        type: 'radar',
        data: {
            labels: aspectos.map(a => a.label),
            datasets: [{
                label: 'M√©dia de Satisfa√ß√£o',
                data: medias,
                backgroundColor: 'rgba(46, 125, 50, 0.2)',
                borderColor: '#2e7d32',
                borderWidth: 2
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            scales: {
                r: {
                    beginAtZero: true,
                    max: 5
                }
            }
        }
    });
}

// 7. ATUALIZAR GR√ÅFICOS AP√ìS FILTRO
function atualizarGraficos() {
    // Destruir gr√°ficos antigos
    Object.values(charts).forEach(chart => chart.destroy());
    charts = {};
    
    // Recriar com dados filtrados
    criarGraficos();
}

// 8. FUN√á√ïES AUXILIARES
function contarFrequencia(campo) {
    const contador = {};
    formulariosFiltrados.forEach(form => {
        const valor = form.respostas[campo];
        if (valor) {
            contador[valor] = (contador[valor] || 0) + 1;
        }
    });
    return contador;
}

function exportarGrafico(chartId) {
    const canvas = document.getElementById(chartId);
    const url = canvas.toDataURL('image/png');
    const a = document.createElement('a');
    a.href = url;
    a.download = `${chartId}.png`;
    a.click();
}

function exportarCSV(dados) {
    const csv = dados.map(row => {
        return row.map(cell => {
            const str = String(cell || '');
            return str.includes(',') ? `"${str}"` : str;
        }).join(',');
    }).join('\n');
    
    const BOM = '\uFEFF';
    const blob = new Blob([BOM + csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'dados.csv';
    a.click();
}

function calcularMedia(campo) {
    const valores = formulariosFiltrados
        .map(f => parseFloat(f.respostas[campo]))
        .filter(v => !isNaN(v));
    
    if (valores.length === 0) return 0;
    
    const soma = valores.reduce((a, b) => a + b, 0);
    return (soma / valores.length).toFixed(2);
}
